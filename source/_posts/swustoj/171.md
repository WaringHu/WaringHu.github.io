# 源代码

```cpp
#include <stdio.h>
#include <string.h>

int main()
{
	char s[100];
	while(~scanf("%s", s))
	{
		int n;
		n=strlen(s);
		for(int i=n-1; i>=0; i--)
		{
			printf("%c", s[i]);
		}
		printf("\n");
	}
	return 0;
}
```
